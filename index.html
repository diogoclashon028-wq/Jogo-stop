const express = require('express'); const http = require('http'); const socketIo = require('socket.io'); const app = express(); const server = http.createServer(app); const io = socketIo(server); // Configuração de porta automática para o Render const PORT = process.env.PORT || 10000; // TODO O SEU CONTEÚDO VISUAL (HTML/CSS/JS DO CLIENTE) const conteudoHTML = ` Painel Dinâmico

Paro! Online

Sala: -----

`; // O servidor entrega o HTML unificado na página inicial app.get('/', (req, res) => { res.send(conteudoHTML); }); let salas = {}; io.on('connection', (socket) => { console.log('Novo jogador conectado:', socket.id); socket.on('criarSala', (nomeSala) => { if (!salas[nomeSala]) { salas[nomeSala] = { jogadores: [], jogoIniciado: false }; socket.join(nomeSala); salas[nomeSala].jogadores.push(socket.id); socket.emit('salaCriada', nomeSala); console.log(`Sala ${nomeSala} criada por ${socket.id}`); } else { // Se a sala já existir, o jogador entra nela em vez de dar erro socket.join(nomeSala); salas[nomeSala].jogadores.push(socket.id); socket.emit('salaCriada', nomeSala); } }); socket.on('disconnect', () => { console.log('Jogador desconectado:', socket.id); }); }); // Inicialização correta para o Render server.listen(PORT, () => { console.log(`Servidor rodando na porta ${PORT}`); });
